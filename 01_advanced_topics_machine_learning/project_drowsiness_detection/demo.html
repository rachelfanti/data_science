cap = cv2.VideoCapture(0)
counter=0 # time counter with eyes closed
if not cap.isOpened():
    print("Cannot open camera")
    exit()
while True:
    # Capture frame-by-frame
    ret, frame = cap.read()
    # if frame is read correctly ret is True
    if not ret:
        print("Can't receive frame (stream end?). Exiting ...")
        break
    # Our operations on the frame come here
    
    #Setup------------------------------------
    
    font = cv2.FONT_HERSHEY_SIMPLEX   
    sp = dlib.shape_predictor("inputs\\shape_predictor_68_face_landmarks.dat") # predictor used to find landmarks
    eyes_pictures = []
    num_pix = 299
            
    # dlib face detection---------------------
    
    img_cvt=cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
      
    detector = dlib.get_frontal_face_detector()
    detections = detector(img_cvt, 1)
    
    if len (detections)==0:
        print ('No face detected')
        status = 'No face detected'
        cv2.putText(frame,status,(50,50),font,1,(0,0,255),2,cv2.LINE_4) 
    else:
        print ('Face detected')
        status = 'Face detected'
        cv2.putText(frame,status,(50,50),font,1,(255,0,0),2,cv2.LINE_4) 
        
        # find landmarks--------------------------
    
        faces = dlib.full_object_detections()
        for det in detections:
            faces.append(sp(frame, det))
    
        # Bounding box for face and eyes--------------------
   
        #face
        bb = [i.rect for i in faces]
        bb = [((i.left(), i.top()), (i.right(), i.bottom())) for i in bb] # Convert out of dlib format
    
        #eyes
        right_eyes = [[face.part(i) for i in range(36, 42)] for face in faces]
        right_eyes = [[(i.x, i.y) for i in eye] for eye in right_eyes] # Convert out of dlib format
    
        left_eyes = [[face.part(i) for i in range(42, 48)] for face in faces]
        left_eyes = [[(i.x, i.y) for i in eye] for eye in left_eyes] # Convert out of dlib format

        if (len(right_eyes)==0 and len(left_eyes)==0):
            print ('Eyes are not detected')
            status = 'No eyes detected'
            cv2.putText(frame, status,(100,100),font,1,(0,0,255),2,cv2.LINE_4)
                        #image, text, org, font, fontScale, color, thickness, cv2.LINE_AA, False)
        else:   
            print ('Eyes detected')
            status = 'Eyes detected'
            cv2.putText(frame, status,(100,100),font,1,(255,0,0),2,cv2.LINE_4)
           
            # Display----------------------------------
    
            #imgd = cv2.cvtColor(frame, cv2.COLOR_RGB2BGR) # Convert back to OpenCV

            for i in bb:
                cv2.rectangle(frame, i[0], i[1], (255,0,0), 2) # Bounding box for face
    
            # Increasing the rectangle:
            
            i= 20

            for eye in right_eyes:
        
                x = round(min(eye, key=lambda x: x[0])[0]-i)
                x_w = round(max(eye, key=lambda x: x[0])[0]+i)
                y = round(max(eye, key=lambda x: x[1])[1]-i)
                y_h = round(min(eye, key=lambda x: x[1])[1]+i)
        
                cv2.rectangle(frame, (x_w, y), (x, y_h), (0, 255, 0), 1) # rectangle: (x+w, y), (x, y+h)
                
                crop = img_cvt[y:y_h, x:x_w] # crop: img[y:y + h, x:x + w]
                if crop is None:
                    print('crop images problems')
                else:
                    crop = cv2.resize(crop,(num_pix,num_pix)) #crop = np.expand_dims(crop, axis = 0) 
                    crop = crop/255
                    eyes_pictures.append(crop)
        
                    
        
                for point in eye:
                    cv2.circle(frame, (point[0], point[1]), 2, (130, 200, 130), -1)

            for eye in left_eyes:
        
                x = round(min(eye, key=lambda x: x[0])[0]-i)
                x_w = round(max(eye, key=lambda x: x[0])[0]+i)
                y = round(max(eye, key=lambda x: x[1])[1]-i)
                y_h = round(min(eye, key=lambda x: x[1])[1]+i)
        
                cv2.rectangle(frame, (x_w, y), (x, y_h), (0,255,0), 1) # rectangle: (x+w, y), (x, y+h)
                
                crop = img_cvt[y:y_h, x:x_w] # crop: img[y:y + h, x:x + w]
                if crop is None:
                    print('crop images problems')
                else:
                    crop = cv2.resize(crop,(num_pix,num_pix)) #crop = np.expand_dims(crop, axis = 0) 
                    crop = incep_v3_preprocess_input(crop)
                    eyes_pictures.append(crop)
                    
                for point in eye:
                    cv2.circle(frame, (point[0], point[1]), 2, (130,200,130), -1)

                #Identifying if eyes are opened or closed----
      
            eyes_pictures = np.array(eyes_pictures)
            
            if len  (eyes_pictures) > 0:
                predictions = model.predict(eyes_pictures)
            
                if min(predictions)>0.5:
                    status = 'Opened eyes'
                    cv2.putText(frame,status,(150,150),font,1,(255,0,0),2,cv2.LINE_4)
                    counter = 0
                else:
                    status = 'Closed_eyes'
                    counter = counter+1
                    cv2.putText(frame,status,(150,150),font,1,(0,0,255),2,cv2.LINE_4)
                    if counter>=2:
                        status = 'ALERT!!!!!'
                        cv2.putText(frame, status,(200,200),font,2,(237,28,36),2,cv2.LINE_AA)
                        winsound.PlaySound('alarm.wav', winsound.SND_FILENAME|winsound.SND_NOWAIT)
                        counter=0
    
    
    # Display the resulting frame
    cv2.imshow("drowsiness detection", frame) 
    
    if cv2.waitKey(1) == ord('q'):
        break

# When everything done, release the capture
cap.release()
cv2.destroyAllWindows()
